#!/usr/bin/env zsh

# Terminal Color Codes
COLOR_RESET='\033[0m'
COLOR_BLACK='\033[30m'
COLOR_RED='\033[31m'
COLOR_GREEN='\033[32m'
COLOR_YELLOW='\033[33m'
COLOR_BLUE='\033[34m'
COLOR_MAGENTA='\033[35m'
COLOR_CYAN='\033[36m'
COLOR_WHITE='\033[37m'

# 256-color functions
color256() {
    echo "\033[38;5;$1m"
}

# Gradient functions
tui_apply_rainbow_gradient() {
    local text="$1"
    local length=${#text}
    local output=""
    for (( i=0; i<length; i++ )); do
        local hue=$(( (i * 360 / length) % 360 ))
        local color_code=$(( 16 + (hue * 5 / 360) * 36 + (hue * 5 / 360 % 6) * 6 + (hue * 5 / 360 / 6) ))
        output+="$(color256 $color_code)${text:$i:1}"
    done
    output+="$COLOR_RESET"
    echo -n "$output"
}

tui_apply_aurora_gradient() {
    local text="$1"
    local length=${#text}
    local output=""
    # Aurora gradient: cyan -> green -> blue -> purple
    for (( i=0; i<length; i++ )); do
        local position=$(( i * 100 / length ))
        local color_code
        if [ $position -lt 25 ]; then
            # Cyan to green
            color_code=$(( 51 - (position * 20 / 25) ))
        elif [ $position -lt 50 ]; then
            # Green to blue
            color_code=$(( 40 + ((position - 25) * 20 / 25) ))
        elif [ $position -lt 75 ]; then
            # Blue to purple
            color_code=$(( 63 + ((position - 50) * 20 / 25) ))
        else
            # Purple to magenta
            color_code=$(( 129 + ((position - 75) * 20 / 25) ))
        fi
        output+="$(color256 $color_code)${text:$i:1}"
    done
    output+="$COLOR_RESET"
    echo -n "$output"
}

# Terminal dimensions
tui_get_terminal_dimensions() {
    terminal_cols=$(tput cols)
    terminal_rows=$(tput lines)
}

tui_calculate_centering() {
    local content_width=$1
    local total_width=$2
    local padding=$(( (total_width - content_width) / 2 ))
    echo $padding
}

tui_pad_center() {
    local text="$1"
    local width=$2
    local text_width=${#text}
    if [ $text_width -ge $width ]; then
        echo "$text"
    else
        local left_padding=$(( (width - text_width) / 2 ))
        local right_padding=$(( width - text_width - left_padding ))
        printf "%${left_padding}s%s%${right_padding}s" "" "$text" ""
    fi
}

tui_repeat_char() {
    local char="$1"
    local count=$2
    printf "%${count}s" "" | tr " " "$char"
}

    local width=$1

# Progress bar
tui_render_progress_bar() {
    local current=$1
    local total=$2
    local width=$3
    local color="$4"
    local percentage=$(( current * 100 / total ))
    local progress=$(( current * width / total ))
    local bar=""
    bar+="${color}[${COLOR_RESET}"
    for ((i=0; i<width; i++)); do
        if [ $i -lt $progress ]; then
            bar+="${color}#${COLOR_RESET}"
        else
            bar+=" "
        fi
    done
    bar+="${color}] ${percentage}%%${COLOR_RESET}"
    echo "$bar"
}

    import { rankRepos } from './ranker.js';
    export async function searchReposByIntent(intent, limit=10) {
      const qParts = [];
      if (intent.language) qParts.push(`language:${intent.language}`);
      if (intent.keywords) qParts.push(intent.keywords.join(' '));
      const q = qParts.join(' ');
      const token = configManager.get('github_token');
      const url = `https://api.github.com/search/repositories?q=${encodeURIComponent(q)}&per_page=${limit}`;
      const res = await axios.get(url, { headers: { Authorization: `token ${token}` }});
      const repos = res.data.items.map(mapRepoToInternal);
      return rankRepos(repos, intent);
    }
